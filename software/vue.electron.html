# npm i -g electron --registry=https://registry.npm.taobao.org
# 改用cnpm安装electron

cnpm install electron -D

vue create election-demo

# add electron-builder
cd election-demo
vue add electron-builder

# retry when fail
cnpm install

# run test

cnpm run electron:serve

# package

cnpm run electron:build

报错Error: Exit code: ENOENT. spawn /usr/bin/python ENOENT

sudo ln -s /usr/bin/python3 /usr/bin/python
Password:
ln: /usr/bin/python: Read-only file system

mac不允许在/usr/bin下建软链接，同时electron:build不支持python3，此时需安装pyenv

pyenv shell 2.7.18
cnpm run electron:build

仍然报错，设置临时PYTHON_PATH

pyenv shell 2.7.18
export PYTHON_PATH=/Users/yan/.pyenv/shims/python && cnpm run electron:build
